struct Pet {

    std::string name;

PYBIND11_MODULE(classs, m) {
    py::class_<Pet>(m, "Pet")

魔法函数

对于python的一些魔法函数,比如 __repr__ 可以借助lambda表达式实现

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    void setName(const std::string &name_) { name = name_; }
    const std::string &getName() const { return name; }
    std::string name;
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
        .def(py::init<const std::string &>())
        .def("setName", &Pet::setName)
        .def("getName", &Pet::getName)
        .def("__repr__",
            [](const Pet &a) {
                return "__repr__ named '" + a.name;
            }
        );
}
import cla

p = cla.Pet("Monika")
print(p)
# __repr__ named 'Monika

并不是所有魔法函数都可以,实际上也没有必要非得去使用

静态字段

可以使用 class_::def_readwrite() 直接暴露字段

也有只读方法 class_::def_readonly()

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    void setName(const std::string &name_) { name = name_; }
    const std::string &getName() const { return name; }
    std::string name;
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
        .def(py::init<const std::string &>())
        .def("setName", &Pet::setName)
        .def("getName", &Pet::getName)
        .def_readwrite("name", &Pet::name);
}
import cla

p = cla.Pet("Monika")
print(p.getName())
p.name = "Jack"
print(p.name)

# Monika
# Jack

private

对于private变量,如果其存在getter和setter也可以使用 class_::def_property() 透明地调用 setter 和 getter 函数

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    void setName(const std::string &name_) { 
        printf("called setter\n");
        name = name_;
    }
    const std::string &getName() const { 
        printf("called getter\n");
        return name;
    }
    std::string name;
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
        .def(py::init<const std::string &>())
        .def_property("name", &Pet::getName, &Pet::setName);
}
import cla

p = cla.Pet("Monika")
p.name = "Jack"
print(p.name)

# called setter
# called getter
# Jack

对于只写属性可以使用nullptr

.def_property("name", nullptr, &Pet::setName);

除此之外还有一些类似的函数用于静态属性

动态属性

python支持动态属性添加,但是pybind11默认不支持,需要添加构造函数 py::dynamic_attr() 标记

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    void setName(const std::string &name_) { name = name_;}
    const std::string &getName() const { return name;}
    std::string name;
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet",py::dynamic_attr())
        .def(py::init<const std::string &>())
        .def_property("name", &Pet::getName, &Pet::setName);
}
import cla

p = cla.Pet("Monika")
p.name = "Jack"
p.age = 18
print(p.name)
print(p.age)

# Jack
# 18
启用动态属性会造成运行耗时增加,不仅仅因为添加了 `__dict__` , 还因为需要激活垃圾回收跟踪.

默认情况下，pybind11 类比本机 Python 类更高效。启用动态属性只会使它们处于同等水平

继承

对于继承关系的构建有两种方式,这里只推荐第一种方式

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    std::string name;
};

struct Dog : Pet {
    Dog(const std::string &name) : Pet(name) { }
    std::string bark() const { return "woof!"; }
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
    .def(py::init<const std::string &>())
    .def_readwrite("name", &Pet::name);
    // Method 1: template parameter:
    py::class_<Dog, Pet>(m, "Dog")
        .def(py::init<const std::string &>())
        .def("bark", &Dog::bark);
}
import cla

p = cla.Dog("Monika")
print(p.name)
print(p.bark())

# Monika
# woof!

多态与虚函数

pybind11种使用多态的情况与C++相同

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name) : name(name) { }
    std::string name;
};

struct Dog : Pet {
    Dog(const std::string &name) : Pet(name) { }
    std::string bark() const { return "woof!"; }
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
    .def(py::init<const std::string &>())
    .def_readwrite("name", &Pet::name);
    // Method 1: template parameter:
    py::class_<Dog, Pet>(m, "Dog")
        .def(py::init<const std::string &>())
        .def("bark", &Dog::bark);
    m.def("pet_store", []() { return std::unique_ptr<Pet>(new Dog("Molly")); });
}
import cla

p = cla.pet_store()
print(type(p))
print(p.name)
print(p.bark())

# output
<class 'cla.Pet'>
Molly
Traceback (most recent call last):
  File "/root/pybind11-code/test.py", line 6, in <module>
    print(p.bark())
AttributeError: 'cla.Pet' object has no attribute 'bark'

该函数返回了一个Dog实例，但因为它是基指针后面的非多态类型. 这里的 std::unique_ptr<Pet> 指针类型无法使用子类的bark函数,在 C++ 中，只有至少有一个虚函数的类型才被认为是多态的，pybind11 会自动识别这一点

添加虚函数即可解决此问题,这与C++的用法一致

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    virtual ~Pet() = default;
    Pet(const std::string &name) : name(name) { }
    std::string name;
};

struct Dog : Pet {
    Dog(const std::string &name) : Pet(name) { }
    std::string bark() const { return "woof!"; }
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
    .def(py::init<const std::string &>())
    .def_readwrite("name", &Pet::name);
    // Method 1: template parameter:
    py::class_<Dog, Pet>(m, "Dog")
        .def(py::init<const std::string &>())
        .def("bark", &Dog::bark);
    m.def("pet_store", []() { return std::unique_ptr<Pet>(new Dog("Molly")); });
}
import cla

p = cla.pet_store()
print(type(p))
print(p.name)
print(p.bark())

# <class 'cla.Dog'>
# Molly
# woof!

重载

对于C++中的重载

struct Pet {
    Pet(const std::string &name, int age) : name(name), age(age) { }
    void set(int age_) { age = age_; }
    void set(const std::string &name_) { name = name_; }
    std::string name;
    int age;
};

直接尝试绑定Pet::set将导致错误，因为编译器不知道用户打算选择哪种方法,可以使用函数指针来消除歧义

#include <pybind11/pybind11.h>

namespace py = pybind11;

struct Pet {
    Pet(const std::string &name, int age) : name(name), age(age) { }
    void set(int age_) { 
        printf("called set of age\n");
        age = age_;
    }
    void set(const std::string &name_) {
        printf("called set of name\n");
        name = name_;
    }
    std::string name;
    int age;
};

PYBIND11_MODULE(cla, m) {
    py::class_<Pet>(m, "Pet")
        .def(py::init<const std::string &, int>())
        .def("set", static_cast<void (Pet::*)(int)>(&Pet::set), "Set the pet's age")
        .def("set", static_cast<void (Pet::*)(const std::string &)>(&Pet::set), "Set the pet's name");
}
import cla

p = cla.Pet("Monika",18)
p.set("Billy")
p.set(20)

# called set of name
# called set of age
